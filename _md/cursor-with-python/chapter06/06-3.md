## 🚀 실습 1: random 모듈 활용하기

### 랜덤 기능 마스터하기

**기본 사용법:**

```python
import random

# 1부터 10 사이의 랜덤 숫자
number = random.randint(1, 10)
print(f"랜덤 숫자: {number}")

# 리스트에서 랜덤 선택
colors = ["빨강", "파랑", "노랑", "초록"]
chosen_color = random.choice(colors)
print(f"선택된 색깔: {chosen_color}")

# 리스트 섞기
cards = ["A", "K", "Q", "J"]
random.shuffle(cards)
print(f"섞인 카드: {cards}")

# 0.0~1.0 사이의 랜덤 실수
probability = random.random()
print(f"확률: {probability:.2f}")
```

### AI와 함께 로또 번호 생성기 만들기

**Chat에서:**

```
random 모듈을 사용해서 로또 번호 생성기를 만들어주세요.

기능:
- 1~45 사이의 숫자 중 6개를 중복 없이 선택
- 번호를 오름차순으로 정렬해서 보여주기
- 여러 번 생성할 수 있는 기능
- 예쁘게 출력하기

초심자가 이해하기 쉽게 만들어주세요.
```

**AI가 만들어준 로또 생성기:**

```python
import random

def generate_lotto_numbers():
    """로또 번호 6개를 생성하는 함수"""
    # 1부터 45까지의 숫자 중 6개를 중복 없이 선택
    numbers = random.sample(range(1, 46), 6)
    # 오름차순으로 정렬
    numbers.sort()
    return numbers

def show_lotto_numbers(numbers):
    """로또 번호를 예쁘게 출력하는 함수"""
    print("🎱 로또 번호: ", end="")
    for i, num in enumerate(numbers):
        if i < len(numbers) - 1:
            print(f"{num:2d}", end=" - ")
        else:
            print(f"{num:2d}")

def lotto_generator():
    """로또 번호 생성기 메인 함수"""
    print("🍀 로또 번호 생성기에 오신 것을 환영합니다!")
    print("=" * 40)
    
    while True:
        print("\n1. 로또 번호 생성")
        print("2. 종료")
        choice = input("선택하세요 (1-2): ")
        
        if choice == "1":
            numbers = generate_lotto_numbers()
            show_lotto_numbers(numbers)
            
            # 보너스 기능: 행운의 색깔도 추천
            colors = ["빨강", "파랑", "노랑", "초록", "보라", "주황"]
            lucky_color = random.choice(colors)
            print(f"🌈 오늘의 행운의 색깔: {lucky_color}")
            
        elif choice == "2":
            print("🍀 행운을 빕니다!")
            break
        else:
            print("❌ 1 또는 2를 선택해주세요!")

# 프로그램 실행
lotto_generator()
```

### 생일 계산기 만들기

**Chat에서:**

```
datetime 모듈을 사용해서 생일 계산기를 만들어주세요.

기능:
- 사용자의 생일 입력받기
- 현재 나이 계산하기
- 다음 생일까지 남은 날 계산하기
- 태어난 요일 알려주기

초심자가 사용하기 쉽게 만들어주세요.
```

**AI가 만들어준 생일 계산기:**

```python
import datetime

def get_birthday():
    """사용자의 생일을 입력받는 함수"""
    while True:
        try:
            print("생일을 입력해주세요.")
            year = int(input("년도 (예: 2000): "))
            month = int(input("월 (1-12): "))
            day = int(input("일 (1-31): "))
            
            birthday = datetime.date(year, month, day)
            return birthday
            
        except ValueError:
            print("❌ 올바른 날짜를 입력해주세요!")

def calculate_age(birthday, today):
    """나이 계산하는 함수"""
    age = today.year - birthday.year
    
    # 생일이 아직 안 지났으면 나이에서 1을 빼기
    if (today.month, today.day) < (birthday.month, birthday.day):
        age -= 1
    
    return age

def days_until_birthday(birthday, today):
    """다음 생일까지 남은 날 계산"""
    # 올해 생일
    this_year_birthday = datetime.date(today.year, birthday.month, birthday.day)
    
    # 올해 생일이 지났으면 내년 생일로
    if this_year_birthday < today:
        next_birthday = datetime.date(today.year + 1, birthday.month, birthday.day)
    else:
        next_birthday = this_year_birthday
    
    # 남은 일수 계산
    days_left = (next_birthday - today).days
    return days_left, next_birthday

def get_birth_weekday(birthday):
    """태어난 요일 구하기"""
    weekdays = ["월요일", "화요일", "수요일", "목요일", "금요일", "토요일", "일요일"]
    return weekdays[birthday.weekday()]

def birthday_calculator():
    """생일 계산기 메인 함수"""
    print("🎂 생일 계산기에 오신 것을 환영합니다!")
    print("=" * 40)
    
    birthday = get_birthday()
    today = datetime.date.today()
    
    # 계산 결과
    age = calculate_age(birthday, today)
    days_left, next_birthday = days_until_birthday(birthday, today)
    birth_weekday = get_birth_weekday(birthday)
    
    # 결과 출력
    print(f"\n🎉 결과")
    print("=" * 30)
    print(f"생일: {birthday.strftime('%Y년 %m월 %d일')}")
    print(f"태어난 요일: {birth_weekday}")
    print(f"현재 나이: {age}세")
    print(f"다음 생일: {next_birthday.strftime('%Y년 %m월 %d일')}")
    
    if days_left == 0:
        print("🎂 오늘이 생일입니다! 생일 축하합니다! 🎉")
    else:
        print(f"다음 생일까지: {days_left}일 남음")

# 프로그램 실행
birthday_calculator()
```

### 수학 계산기 만들기

**Chat에서:**

```
math 모듈을 사용해서 고급 계산기를 만들어주세요.

기능:
- 제곱, 제곱근, 팩토리얼 계산
- 원의 넓이, 둘레 계산
- 삼각함수 계산 (sin, cos, tan)
- 로그 계산

메뉴 방식으로 만들어주시고 초심자도 쉽게 사용할 수 있게 해주세요.
```

**AI가 만들어준 고급 계산기:**

```python
import math

def power_calculator():
    """거듭제곱 계산"""
    base = float(input("밑: "))
    exponent = float(input("지수: "))
    result = math.pow(base, exponent)
    print(f"{base}의 {exponent}승 = {result}")

def sqrt_calculator():
    """제곱근 계산"""
    number = float(input("숫자 입력: "))
    if number < 0:
        print("❌ 음수의 제곱근은 계산할 수 없습니다!")
        return
    result = math.sqrt(number)
    print(f"√{number} = {result}")

def factorial_calculator():
    """팩토리얼 계산"""
    number = int(input("숫자 입력 (정수): "))
    if number < 0:
        print("❌ 음수의 팩토리얼은 계산할 수 없습니다!")
        return
    result = math.factorial(number)
    print(f"{number}! = {result}")

def circle_calculator():
    """원의 넓이와 둘레 계산"""
    radius = float(input("반지름 입력: "))
    area = math.pi * radius ** 2
    circumference = 2 * math.pi * radius
    print(f"원의 넓이: {area:.2f}")
    print(f"원의 둘레: {circumference:.2f}")

def trigonometry_calculator():
    """삼각함수 계산"""
    angle_degrees = float(input("각도 입력 (도): "))
    angle_radians = math.radians(angle_degrees)  # 도를 라디안으로 변환
    
    sin_value = math.sin(angle_radians)
    cos_value = math.cos(angle_radians)
    tan_value = math.tan(angle_radians)
    
    print(f"sin({angle_degrees}°) = {sin_value:.4f}")
    print(f"cos({angle_degrees}°) = {cos_value:.4f}")
    print(f"tan({angle_degrees}°) = {tan_value:.4f}")

def advanced_calculator():
    """고급 계산기 메인 함수"""
    print("🔢 고급 계산기")
    print("=" * 30)
    
    while True:
        print("\n계산할 기능을 선택하세요:")
        print("1. 거듭제곱 (x^y)")
        print("2. 제곱근 (√x)")
        print("3. 팩토리얼 (x!)")
        print("4. 원의 넓이와 둘레")
        print("5. 삼각함수 (sin, cos, tan)")
        print("6. 종료")
        
        choice = input("선택 (1-6): ")
        
        try:
            if choice == "1":
                power_calculator()
            elif choice == "2":
                sqrt_calculator()
            elif choice == "3":
                factorial_calculator()
            elif choice == "4":
                circle_calculator()
            elif choice == "5":
                trigonometry_calculator()
            elif choice == "6":
                print("👋 계산기를 종료합니다!")
                break
            else:
                print("❌ 1~6 중에서 선택해주세요!")
                
        except ValueError:
            print("❌ 올바른 숫자를 입력해주세요!")

# 프로그램 실행
advanced_calculator()
```

**Before (모듈 없이):**
- 모든 기능을 직접 구현해야 함
- 복잡한 계산이나 처리가 어려움
- 개발 시간이 오래 걸림

**After (모듈 활용):**
- 검증된 기능들을 바로 사용
- 복잡한 기능도 쉽게 구현
- 개발 속도가 획기적으로 빨라짐
