### 🎯 실습 1: 나만의 코딩 패턴 만들기

**1단계: 내 코딩 스타일 정의하기**

```python
# 나의 표준 에러 처리 패턴
def safe_input_number(prompt, min_val=0, max_val=100):
    """안전한 숫자 입력 - 내 표준 패턴"""
    while True:
        try:
            user_input = input(prompt)
            if user_input.isdigit():
                number = int(user_input)
                if min_val <= number <= max_val:
                    return number
                else:
                    print(f"❌ {min_val}~{max_val} 사이의 숫자를 입력하세요.")
            else:
                print("❌ 숫자를 입력하세요.")
        except KeyboardInterrupt:
            print("\n프로그램을 종료합니다.")
            return None

# 나의 표준 데이터 출력 패턴
def display_data_with_header(title, data_list):
    """표준 데이터 출력 - 내 스타일"""
    print(f"\n{'='*50}")
    print(f"📊 {title}")
    print(f"{'='*50}")
    
    if not data_list:
        print("표시할 데이터가 없습니다.")
        return
    
    for i, item in enumerate(data_list, 1):
        if isinstance(item, dict):
            # 딕셔너리인 경우
            name = item.get('name', '이름없음')
            score = item.get('score', 0)
            grade = item.get('grade', 'N/A')
            print(f"{i:2d}. {name}: {score}점 ({grade}등급)")
        else:
            # 기타인 경우
            print(f"{i:2d}. {item}")
    
    print(f"{'='*50}")
```

display_data_with_header: 데이터를 예쁘게 출력하는 제 표준 방식

```python
# 내가 자주 쓰는 메뉴 시스템 패턴
class MenuSystem:
    def __init__(self, title):
        self.title = title
        self.options = {}
    
    def add_option(self, key, description, function):
        """메뉴 옵션 추가"""
        self.options[key] = {
            "description": description,
            "function": function
        }
    
    def show_menu(self):
        """메뉴 출력"""
        print(f"\n{'='*50}")
        print(f"🎯 {self.title}")
        print(f"{'='*50}")
        
        for key, option in self.options.items():
            print(f"{key}. {option['description']}")
        
        print("0. 종료")
        print(f"{'='*50}")
    
    def run(self):
        """메뉴 실행"""
        while True:
            self.show_menu()
            choice = input("\n선택하세요: ")
            
            if choice == "0":
                print("프로그램을 종료합니다.")
                break
            elif choice in self.options:
                try:
                    self.options[choice]["function"]()
                except Exception as e:
                    print(f"❌ 오류가 발생했습니다: {e}")
            else:
                print("❌ 올바른 메뉴를 선택해주세요.")
```
