### 🎨 종합 실습 1: 성적 관리 시스템

AI가 만들어줄 코드:

```python
def main():
    print("=== 반 전체 성적 관리 시스템 ===")
    
    # 학생 정보 리스트
    students = ["김철수", "이영희", "박민수", "최지영", "정수호"]
    scores = [85, 92, 78, 95, 88]
    
    # 모든 학생 성적 출력
    print("\n📊 학생별 성적:")
    for i in range(len(students)):
        print(f"{students[i]}: {scores[i]}점")
    
    # 평균 성적 계산
    total = 0
    for score in scores:
        total += score
    average = total / len(scores)
    
    print(f"\n📈 반 평균: {average:.1f}점")
    
    # 최고점, 최저점 찾기
    max_score = max(scores)
    min_score = min(scores)
    
    print(f"🏆 최고점: {max_score}점")
    print(f"📉 최저점: {min_score}점")
    
    # 평균 이상 학생 찾기
    print(f"\n🌟 평균 이상 학생들:")
    for i in range(len(students)):
        if scores[i] >= average:
            print(f"👏 {students[i]}: {scores[i]}점")

```

    if condition(item):
        process(item)

# 3. 번호와 함께 처리

    print(f"{i+1}번째: {items[i]}")
